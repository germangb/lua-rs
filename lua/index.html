extern crate lua;

use lua::Index;

let mut state = lua::State::new();
state.open_libs();

state.eval(r#"
    print ('hello world')

    -- define global vars
    foo = 42
"#).unwrap();

state.get_global("foo").unwrap();

assert_eq!(Ok(42), state.get(-1));
assert_eq!(Ok("42"), state.get(-1));
